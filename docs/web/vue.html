    //1 上方這種一般的函式，可以被改寫成下方的箭頭函式
    elements.map((element) => {
    return element.length;
    }); // [8, 6, 7, 9]

    //2 如果輸入的參數只有一個，我們可以移除掉外面的括號
    elements.map(element => {
    return element.length;
    }); // [8, 6, 7, 9]

    //3 當箭頭函式裡的內容只有'return'的時候，我們可以拿掉return+外面的大括號
    elements.map((element) =>  element.length); // [8, 6, 7, 9]
    elements.map(element) =>  element.length); // [8, 6, 7, 9]
